```python
import pandas as pd

class BaseStrategy(ABC):
    """策略基类"""
    
    def __init__(self, config: dict):
        self.config = config
        self.name = "BaseStrategy"
        
    @abstractmethod
    def analyze(self, df: pd.DataFrame) -> pd.DataFrame:
        """分析数据，返回信号"""
        pass
    
    @abstractmethod
    def generate_signals(self, df: pd.DataFrame) -> pd.DataFrame:
        """生成交易信号"""
        pass
    
    def validate_signal(self, signal: dict) -> bool:
        """验证信号有效性 (风控)"""
        pass


class BreakoutStrategy(BaseStrategy):
    """突破策略 - 经典技术分析"""
    
    def __init__(self, config: dict):
        super().__init__(config)
        self.name = "突破策略"
        self.lookback = config.get("lookback_period", 20)
        self.threshold = config.get("threshold", 0.02)
    
    def analyze(self, df: pd.DataFrame) -> pd.DataFrame:
        # 计算指标
        df['highest'] = df['high'].rolling(self.lookback).max()
        df['lowest'] = df['low'].rolling(self.lookback).min()
        df['atr'] = self.calculate_atr(df)  # 真实波幅
        
        return df
    
    def generate_signals(self, df: pd.DataFrame) -> pd.DataFrame:
        # 突破买入信号
        df['signal'] = 0
        df.loc[df['close'] > df['highest'].shift(1), 'signal'] = 1  # 多
        df.loc[df['close'] < df['lowest'].shift(1), 'signal'] = -1  # 空
        return df
```

### 2.4 风控层 (Risk Management)

这是最重要的层！参考 freqtrade 设计：

```python
class RiskManager:
    """风控管理器"""
    
    def __init__(self, config: dict):
        # 仓位管理
        self.max_position_pct = config.get("max_position_pct", 0.1)  # 最大10%仓位
        self.max_leverage = config.get("max_leverage", 10)
        
        # 止损止盈
        self.stop_loss_pct = config.get("stop_loss_pct", 0.02)  # 2%止损
        self.take_profit_pct = config.get("take_profit_pct", 0.05)  # 5%止盈
        
        # 单日限制
        self.max_daily_loss_pct = config.get("max_daily_loss_pct", 0.1)  # 单日最多亏10%
        self.max_trades_per_day = config.get("max_trades_per_day", 10)
        
        # 熔断
        self.circuit_breaker_loss_pct = config.get("circuit_breaker_loss_pct", 0.2)  # 亏20%熔断
        
    def check_position_size(self, signal: dict, balance: float) -> float:
        """计算仓位大小"""
        max_size = balance * self.max_position_pct
        return min(signal.get("size", max_size), max_size)
    
    def check_stop_loss(self, entry_price: float, current_price: float, side: str) -> bool:
        """检查是否触发止损"""
        if side == "long":
            loss_pct = (entry_price - current_price) / entry_price
        else:
            loss_pct = (current_price - entry_price) / entry_price
            
        return loss_pct >= self.stop_loss_pct
    
    def check_take_profit(self, entry_price: float, current_price: float, side: str) -> bool:
        """检查是否触发止盈"""
        if side == "long":
            profit_pct = (current_price - entry_price) / entry_price
        else:
            profit_pct = (entry_price - current_price) / entry_price
            
        return profit_pct >= self.take_profit_pct
    
    def check_circuit_breaker(self, daily_pnl_pct: float) -> bool:
        """检查是否熔断"""
        return daily_pnl_pct <= -self.circuit_breaker_loss_pct
    
    def should_trade(self, daily_stats: dict) -> bool:
        """检查是否可以交易"""
        # 检查单日亏损
        if daily_stats.get("pnl_pct", 0) <= -self.max_daily_loss_pct:
            return False
            
        # 检查交易次数
        if daily_stats.get("trade_count", 0) >= self.max_trades_per_day:
            return False
            
        return True
```

## 四、策略选择建议

### 市场环境 vs 策略

| 市场环境 | 推荐策略 | 原因 |
|----------|----------|------|
| 趋势明显 | 突破策略 | 顺势而为 |
| 震荡整理 | 网格策略 | 高卖低买 |
| 波动剧烈 | 突破+风控 | 严格止损 |
| 数据驱动 | ML策略 | AI自适应 |
